            self.start2[i][j] = int(self.start2[i][j])
            
        k = len(self.start)
        for k in range(k) :
            self.finish2[i][k] = int(self.finish2[i][k])
                
        
        # 각 유닉의 이동 거리 및 도달 시간 계산
        self.distance = distance

        l = len(self.start2)

            # 이동 거리 : 피타고라스의 정리("c^2 = a^2 + b^2") 활용
